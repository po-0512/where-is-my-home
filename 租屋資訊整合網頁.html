       // --- 3. 距離與時間計算 (Google API Mock) ---
       // 實務上這裡會呼叫 Distance Matrix API，此處用 Haversine 公式配合速率模擬
       function calculateDistances(rentals) {
           const isPeak = document.getElementById('commuteScenario').value === 'peak';
           
           rentals.forEach(rental => {
               rental.commuteData = [];
               let totalScore = 0;
               let totalWeight = 0;

               refPoints.forEach(ref => {
                   // 計算直線距離 (模擬 API 回傳的距離)
                   const R = 6371; // km
                   const dLat = (ref.lat - rental.lat) * Math.PI / 180;
                   const dLon = (ref.lng - rental.lng) * Math.PI / 180;
                   const a = Math.sin(dLat/2)*Math.sin(dLat/2) + Math.cos(rental.lat*Math.PI/180)*Math.cos(ref.lat*Math.PI/180)*Math.sin(dLon/2)*Math.sin(dLon/2);
                   const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
                   const distanceKm = +(R * c * 1.2).toFixed(1); // 乘 1.2 模擬實際道路彎曲

                   // 估算時間區間 (模擬 API 回傳)
                   const speed = isPeak ? 25 : 40; // 尖峰時速25km/h，離峰40km/h
                   const baseMin = Math.round((distanceKm / speed) * 60) + 5; // +5分找車位/等紅綠燈
                   const minTime = baseMin;
                   const maxTime = Math.round(baseMin * (isPeak ? 1.5 : 1.2));

                   rental.commuteData.push({
                       refName: ref.name,
                       distance: distanceKm,
                       timeStr: `約 ${minTime}–${maxTime} 分鐘`,
                       avgTime: (minTime + maxTime) / 2
                   });

                   // 加權計算 (以平均時間為基底)
                   totalScore += ((minTime + maxTime) / 2) * ref.weight;
                   totalWeight += ref.weight;
               });
               
               // 綜合通勤分數 (越低越好)
               rental.score = totalWeight > 0 ? (totalScore / totalWeight).toFixed(1) : 0;
           });
       }

       function applyFilters() {
           // 同步主地點資料
           refPoints[0].name = document.getElementById('mainRefName').value || '主要地點';
           refPoints[0].address = document.getElementById('mainRefAddress').value;
           renderRefPointsSettings();

           // 取得條件
           const minP = parseInt(document.getElementById('priceMin').value) || 0;
           const maxP = parseInt(document.getElementById('priceMax').value) || 999999;
           const type = document.getElementById('roomType').value;
           const sub = document.getElementById('filterSubsidy').checked;
           const agt = document.getElementById('filterAgent').checked;
           const region = document.getElementById('searchRegion').value;

           // 產生標籤 UI
           const tags = [`<span class="px-2 py-1 bg-blue-100 text-blue-700 text-xs rounded-full">地區: ${region}</span>`];
           if(minP > 0 || maxP < 999999) tags.push(`<span class="px-2 py-1 bg-gray-200 text-gray-700 text-xs rounded-full">預算: ${minP}-${maxP}</span>`);
           if(type !== 'all') tags.push(`<span class="px-2 py-1 bg-gray-200 text-gray-700 text-xs rounded-full">房型: ${type}</span>`);
           if(sub) tags.push(`<span class="px-2 py-1 bg-green-100 text-green-700 text-xs rounded-full">可補助</span>`);
           if(agt) tags.push(`<span class="px-2 py-1 bg-gray-200 text-gray-700 text-xs rounded-full">屋主自租</span>`);
           document.getElementById('activeTags').innerHTML = tags.join('');

           // 執行篩選
           filteredRentals = rawRentals.filter(r => {
               if(r.price < minP || r.price > maxP) return false;
               if(type !== 'all' && r.type !== type) return false;
               if(sub && !r.subsidy) return false;
               if(agt && r.agent) return false;
               // 簡易地區字串比對
               if(!r.location.includes(region.replace('市','').replace('區',''))) return false;
               return true;
           });

           // 收起進階選單
           document.getElementById('advancedFilters').classList.remove('open');
           document.getElementById('advIcon').textContent = '▼';

           // 重新計算通勤距離與分數
           calculateDistances(filteredRentals);
           
           // 渲染
           renderList();
           renderMockMap();
       }

       function renderList() {

           if(sortMethod === 'priceAsc') sorted.sort((a,b) => a.price - b.price);
           if(sortMethod === 'priceDesc') sorted.sort((a,b) => b.price - a.price);
